# ============================================================
# FUNCIONES
# ============================================================

def noop() -> None: {}

def myAdd(a: int, b: int) -> int: {
    return a + b;
}

def double_it(n: int) -> int: {
    return myAdd(n, n);
}

def quad(n: int) -> int: {
    return double_it(double_it(n));
}

def myMod(a: int, b: int) -> int: {
    return a % b;
}

# Todos los operadores aritméticos
def arithTest(a: int, b: int) -> None: {
    r: int;
    r = a + b; print r, ' ';
    r = a - b; print r, ' ';
    r = a * b; print r, ' ';
    r = a / b; print r, ' ';
    r = a % b; print r, '\n';
}

# Todos los operadores de comparación
def cmpTest(a: int, b: int) -> None: {
    print a < b,  ' ';
    print a > b,  ' ';
    print a <= b, ' ';
    print a >= b, ' ';
    print a == b, ' ';
    print a != b, '\n';
}

# Operadores lógicos con int
def logicTest(a: int, b: int) -> None: {
    print a && b, ' ';
    print a || b, ' ';
    print !a,     ' ';
    print !b,     '\n';
}

# Aritmética mixta: char, int, double
def mixedArith() -> None: {
    i: int;
    d: double;
    c: char;
    i = 10;
    d = 2.5;
    c = 'A';
    print (int)(i * d), ' ';
    print c + i, ' ';
    print c - 'a', '\n';
}

# Todos los casts
def castTest() -> None: {
    i: int;
    d: double;
    c: char;
    i = 65;
    c = (char)i;   print c, ' ';
    d = (double)i;
    i = (int)d;    print i, ' ';
    i = (int)3.9;  print i, ' ';
    c = (char)48;  print c, '\n';
}

# Aritmética double
def scaleD(x: double, f: double) -> double: {
    return x * f;
}

# Comparación double
def cmpDouble(a: double, b: double) -> None: {
    print a < b,  ' ';
    print a > b,  ' ';
    print a <= b, ' ';
    print a >= b, '\n';
}

# Promoción implícita char en todas las operaciones
def charOps(c: char) -> None: {
    print c + 0,  ' ';
    print !c,     ' ';
    print -c,     '\n';
}

# charAsInt: return char implicitly promotes to int
def charAsInt(c: char) -> int: {
    return c;
}

# andOf char && char
def andOf(a: char, b: char) -> int: {
    return a && b;
}

# orOf: char || char  (prueba b2i antes de or — paralelo a andOf)
def orOf(a: char, b: char) -> int: {
    return a || b;
}

# Fibonacci iterativo
def fibonacci(n: int) -> int: {
    a, b, temp, i: int;
    a = 0;
    b = 1;
    i = 0;
    while i < n: {
        temp = a + b;
        a = b;
        b = temp;
        i = i + 1;
    }
    return a;
}

# Max de dos enteros (if/else con return en ambas ramas)
def max(a: int, b: int) -> int: {
    if a > b:
        return a;
    else:
        return b;
}

# Clamp: if/else anidado
def clamp(val: int, lo: int, hi: int) -> int: {
    if val < lo: {
        return lo;
    }
    else: {
        if val > hi: {
            return hi;
        }
        else: {
            return val;
        }
    }
}

# Rellena arr1
def fillArr1() -> None: {
    i: int;
    i = 0;
    while i < 5: {
        arr1[i] = i * i;
        i = i + 1;
    }
}

# Rellena arr2
def fillArr2() -> None: {
    i, j: int;
    i = 0;
    while i < 3: {
        j = 0;
        while j < 4: {
            arr2[i][j] = i * 4 + j;
            j = j + 1;
        }
        i = i + 1;
    }
}

# Rellena array de struct
def fillPoints() -> None: {
    i: int;
    i = 0;
    while i < 4: {
        points[i].x = i * 2;
        points[i].y = i * 2 + 1;
        i = i + 1;
    }
}

# Rellena struct con campo array
def fillBag() -> None: {
    bag.data[0] = 10;
    bag.data[1] = 20;
    bag.data[2] = 30;
    bag.size = 3;
}

# promoArgs: conversión implícita en argumentos de función
# — char→int al pasar a myAdd (espera int)
# — int→double al pasar a scaleD (espera double)
# — char→double en asignación directa sin cast
def promoArgs() -> None: {
    d: double;
    print myAdd('A', 1), '\n';
    print (int)scaleD(3, 2), '\n';
    d = 'A';
    print (int)d, '\n';
}

# mixedCmp: comparación entre tipos distintos (superType + conversión antes de comparar)
# int < double  → i2f sobre el int, luego ltf
# char < int    → b2i sobre el char, luego lti
def mixedCmp() -> None: {
    print 3 < 3.5,   ' ';
    print 4 > 3.5,   ' ';
    print 'A' < 100, ' ';
    print 'z' > 100, '\n';
}
